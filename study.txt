package kata

func ValidParentheses(parens string) bool {
  count := 0
  
  for _, element := range parens{
    if string(element) == "("{
      count++
    }

    if string(element) == ")"{
      if count <= 0 {
        return false
      }
      count--
    }
  }
  
  if count == 0{
    return true
  }else{
    return false
  }
}
